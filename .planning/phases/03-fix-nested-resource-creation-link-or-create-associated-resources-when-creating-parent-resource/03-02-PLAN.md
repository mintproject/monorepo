if (junctionDeletions.length === 0 && Object.keys(junctionInserts).length === 0) {

} else {
  // Build multi-root mutation: delete junctions -> update scalars -> insert junctions
  // Hasura executes multiple root fields sequentially in a single transaction (D-03)
  const parts: string[] = []
  const variables: Record<string, unknown> = { id: fullId, set: input }

  // Step 1: Delete existing junction rows for each relationship in the request
  for (let i = 0; i < junctionDeletions.length; i++) {
    const { table, parentFkColumn } = junctionDeletions[i]
    const tableSuffix2 = table.replace('modelcatalog_', '')
    parts.push(`del_${i}: delete_modelcatalog_${tableSuffix2}(where: { ${parentFkColumn}: { _eq: $id } }) { affected_rows }`)

  // Step 2: Update scalar columns

  // Step 3: Insert new junction rows for each relationship

    const relEntry = Object.values(resourceConfig.relationships).find(r => r.hasuraRelName === hasuraRelName)

    const varName = `junc_${hasuraRelName}`

    const juncData = insertData as { data: Record<string, unknown>[] }
    variables[varName] = juncData.data.map((row: Record<string, unknown>) => ({

**IMPORTANT per Pitfall 2:** Only delete junction rows for relationships whose API field name is PRESENT in `body` (even if value is `[]` meaning "clear"). If a relationship field is NOT in the request body, do NOT touch its junction rows.

**IMPORTANT:** The junction insert objects at the root level of a multi-root insert mutation use flat FK columns, NOT nested relationship syntax. So for inserting into `modelcatalog_software_category`, the objects are `[{ software_id: $id, category_id: "..." }]` NOT the nested `{ category: { data: {...} } }` format. The nested format is only for nested inserts WITHIN a parent insert (used in create()). For root-level `insert_modelcatalog_software_category(objects: ...)`, use flat FK columns.

// For each junction relationship in the body, build flat insert objects
for (const [apiFieldName, relConfig] of Object.entries(resourceConfig.relationships)) {
  if (!relConfig.junctionTable || !relConfig.junctionRelName || !relConfig.parentFkColumn) continue
  if (body[apiFieldName] === undefined) continue

  const rawValue = body[apiFieldName]
  if (!Array.isArray(rawValue) || rawValue.length === 0) continue

  const items = rawValue.map((item: unknown) => typeof item === 'string' ? { id: item } : item as Record<string, unknown>)
  const targetFkColumn = `${relConfig.junctionRelName}_id`
  const juncSuffix = relConfig.junctionTable.replace('modelcatalog_', '')
  const varName = `junc_${relConfig.hasuraRelName}`

  parts.push(`ins_${relConfig.hasuraRelName}: insert_modelcatalog_${juncSuffix}(objects: $${varName}, on_conflict: { constraint: modelcatalog_${juncSuffix}_pkey, update_columns: [] }) { affected_rows }`)

  variables[varName] = items.map((item: Record<string, unknown>) => {
    const rawId = item['id'] as string | undefined
    const targetId = rawId
      ? (rawId.startsWith('https://') ? rawId : `https://w3id.org/okn/i/mint/${rawId}`)
      : `https://w3id.org/okn/i/mint/${randomUUID()}`
    return {
      [relConfig.parentFkColumn!]: fullId,

    }
  })
}
