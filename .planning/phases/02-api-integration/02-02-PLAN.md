function resolveToW3Id(urlOrW3Id: string): string {
    if (urlOrW3Id.startsWith("https://w3id.org/")) return urlOrW3Id;
    // REST API URL or short ID: extract trailing segment
    const base = urlOrW3Id.split("?")[0];
    const segment = base.split("/").pop();
    return W3_ID_URI_PREFIX + segment;
}
```

export const fetchModelConfigurationSetup = async (
    url: string
): Promise<CustomModelConfigurationSetup> => {
    const w3Id = resolveToW3Id(url);
    const APOLLO_CLIENT = GraphQL.instance(KeycloakAdapter.getUser());
    const result = await APOLLO_CLIENT.query({
        query: GET_SETUP_QUERY,
        variables: { id: w3Id },
        fetchPolicy: "no-cache"
    });
    const rows = result.data?.modelcatalog_model_configuration_setup ?? [];
    if (rows.length === 0) {
        throw new Error(`ModelConfigurationSetup not found: ${w3Id}`);
    }
    return setupFromGraphQL(rows[0]);
};
```

10. **Rewrite `fetchModelFromCatalog()`:**
The original: calls REST `custom/modelconfigurationsetups/variable?label=...`, iterates results to find matching modelid, then calls `custom/modelconfigurationsetups/{id}` for the matched setup.
New version: query `modelcatalog_model_configuration_setup` with broad search, iterate to find matching id suffix, then fetch the full setup:
```typescript
export const fetchModelFromCatalog = async (
    response_variables: string[],
    driving_variables: string[],
    modelid: string,
    prefs: MintPreferences
): Promise<ModelConfigurationSetup> => {
    const APOLLO_CLIENT = GraphQL.instance(KeycloakAdapter.getUser());
    const result = await APOLLO_CLIENT.query({
        query: GET_SETUPS_BY_VARIABLE_QUERY,
        variables: { label: `%${response_variables.join(",")}%` },
        fetchPolicy: "no-cache"
    });
    const setups = result.data?.modelcatalog_model_configuration_setup ?? [];
    for (const setup of setups) {
        const setupName = setup.id.replace(/.*\//, "");
        if (setupName === modelid) {
            console.log("We found a matching model: " + setup.id + ". Get details");
            return fetchModelConfigurationSetup(setup.id);
        }
    }
    throw new Error(`Model not found: ${modelid}`);
};
```
